#Check port:
netstat -aof | findstr 5432

# Rodar container Postgres na versão 16.3
# Já estou usando a porta 5432 para outro projeto, então apontei para 5433
sudo docker run --name librarydb -e POSTGRES_USER=postgres -e POSTGRES_PASSWORD=postgres -e POSTGRES_DB=library -p 5433:5432 ${-d se quiser modo interativo, fica rodando mesmo se fechar o terminal} postgres:16.3

# Rodar Pgadmin 4
#Esse não usei, usei o dbeaver
sudo docker run --name pgadmin4 -e PGADMIN_DEFAULT_EMAIL=admin@admin.com -e PGADMIN_DEFAULT_PASSWORD=admin -p 15432:80 dpage/pgadmin4:8.9

# Criando uma network
sudo docker network create library-network
## posso recriar os containers acima apontando para o network
sudo docker run --name librarydb -e POSTGRES_USER=postgres -e POSTGRES_PASSWORD=postgres -e POSTGRES_DB=library -p 5433:5432 --network ${library-network} postgres:16.3
sudo docker run --name pgadmin4 -e PGADMIN_DEFAULT_EMAIL=admin@admin.com -e PGADMIN_DEFAULT_PASSWORD=admin -p 15432:80 --network ${library-network} dpage/pgadmin4:8.9
