# Rodar container Postgres na versão 16.3
# Já estou usando a porta 5432 para outro projeto, então apontei para 5433
sudo docker run --name librarydb -e POSTGRES_USER=postgres -e POSTGRES_PASSWORD=postgres -e POSTGRES_DB=library -p 5433:5432 postgres:16.3
